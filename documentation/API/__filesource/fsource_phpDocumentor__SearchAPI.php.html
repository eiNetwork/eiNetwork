  84.  
  85.         if ($searchObject->getResultTotal() < 1) {

  88.             $jsonResults['recordCount'] = 0;

  91.             $error = $searchObject->getIndexError();

  97.                     $jsonResults['parseError'] = true;
  98.

  100.                 } else {
  101.                     PEAR::raiseError(new PEAR_Error('Unable to process query<br />' .
  102.                         'Solr Returned: ' . $error));
  103.                 }
  104.             }
  105.  
  106.             $timer->logTime('no hits processing');
  107.  
  108.         } else {
  109.             $timer->logTime('save search');
  110.  
  111.             // Assign interface variables
  112.             $summary = $searchObject->getResultSummary();
  113.             $jsonResults['recordCount'] = $summary['resultTotal'];
  114.             $jsonResults['recordStart'] = $summary['startRecord'];
  115.             $jsonResults['recordEnd'] =   $summary['endRecord'];
  116.  
  117.             // Big one - our results
  118.             $recordSet = $searchObject->getResultRecordSet();
  119.             //Remove fields as needed to improve the display.
  120.             foreach($recordSet as $recordKey => $record){
  121.                 unset($record['fullrecord']);
  122.                 unset($record['auth_author']);
  123.                 unset($record['auth_authorStr']);
  124.                 unset($record['callnumber-first-code']);
  125.                 unset($record['spelling']);

  127.                 unset($record['title_auth']);
  128.                 unset($record['callnumber-subject']);
  129.                 unset($record['author-letter']);
  130.                 unset($record['marc_error']);
  131.                 unset($record['title_fullStr']);
  132.                 unset($record['shortId']);
  133.                 $recordSet[$recordKey] = $record;
  134.             }
  135.             $jsonResults['recordSet'] = $recordSet;
  136.             $timer->logTime('load result records');
  137.  
  138.             $facetSet = $searchObject->getFacetList();
  139.             $jsonResults['facetSet'] =       $facetSet;
  140.  
  141.             //Check to see if a format category is already set
  142.             $categorySelected = false;
  143.             if (isset($facetSet['top'])){
  144.                 foreach ($facetSet['top'] as $title=>$cluster){
  145.                     if ($cluster['label'] == 'Category'){
  146.                         foreach ($cluster['list'] as $thisFacet){
  147.                             if ($thisFacet['isApplied']){
  148.                                 $categorySelected = true;
  149.                             }
  150.                         }
  151.                     }
  152.                 }
  153.             }
  154.             $jsonResults['categorySelected'] = $categorySelected;
  155.             $timer->logTime('load selected category');
  156.  
  157.             // Process Paging
  158.             $link = $searchObject->renderLinkPageTemplate();
  159.             $options = array('totalItems' => $summary['resultTotal'],
  160.                              'fileName'   => $link,
  161.                              'perPage'    => $summary['perPage']);
  162.             $pager = new VuFindPager($options);
  163.             $jsonResults['paging'] = array(
  164.                 'currentPage' => $pager->pager->_currentPage,
  165.                 'totalPages' => $pager->pager->_totalPages,
  166.                 'totalItems' => $pager->pager->_totalItems,
  167.                 'itemsPerPage' => $pager->pager->_perPage,
  168.             );
  169.             $interface->assign('pageLinks', $pager->getLinks());
  170.             $timer->logTime('finish hits processing');
  171.         }
  172.  
  173.         // Report additional information after the results
  174.         $jsonResults['query_time'] =           round($searchObject->getQuerySpeed(), 2);
  175.         $jsonResults['spellingSuggestions'] = $searchObject->getSpellingSuggestions();

  184.         $jsonResults['showSaved'] =   true;
  185.         $jsonResults['savedSearch'] = $searchObject->isSavedSearch();
  186.         $jsonResults['searchId'] =    $searchObject->getSearchId();
  187.         $currentPage = isset($_REQUEST['page']) ? $_REQUEST['page'] : 1;
  188.         $jsonResults['page'] = $currentPage;
  189.  
  190.  
  191.         if ($configArray['Statistics']['enabled'] && isset( $_GET['lookfor'])) {
  192.             require_once('Drivers/marmot_inc/SearchStat.php');
  193.             $searchStat = new SearchStat();
  194.             $searchStat->saveSearch( strip_tags($_GET['lookfor']), strip_tags($_GET['type']), $searchObject->getResultTotal());
  195.         }
  196.  
  197.         // Save the ID of this search to the session so we can return to it easily:

  200.         // Save the URL of this search to the session so we can return to it easily:
  201.         $_SESSION['lastSearchURL'] = $searchObject->renderSearchUrl();
  202.  
  203.         // Return the results for display to the user.
  204.         return $jsonResults;
  205.     }
